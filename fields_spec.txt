# Конфигурация полей для отображения списка пользователей
# Каждая строка определяет одно поле в одном из форматов:
#
# Формат 1: поле
#   Пример: email
#   Использует имя поля как заголовок, ширина 15, добавляется в конец
#
# Формат 2: поле:ширина
#   Пример: email:25
#   Использует имя поля как заголовок, указанная ширина, добавляется в конец
#
# Формат 3: поле:заголовок:ширина
#   Пример: email:Email:25
#   Пользовательский заголовок, указанная ширина, добавляется в конец
#
# Формат 4: поле:заголовок:ширина:позиция
#   Пример: nickname:Логин:15:1
#   Пользовательский заголовок, ширина, точная позиция в таблице
#
# Пустые строки и строки начинающиеся с # игнорируются

# Основные поля (с русскими заголовками)
nickname:Логин:15:1
name.first:Имя:15:2
name.last:Фамилия:15:3
isAdmin:Админ:8:4
isEnabled:Активен:9:5
